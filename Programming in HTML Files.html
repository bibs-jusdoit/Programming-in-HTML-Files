Third program

To Check Vowel or consonant

CODE

#include <stdio.h>
int main() {
    char c;
    int lowercase_vowel, uppercase_vowel;
    printf("Enter an alphabet: ");
    scanf("%c", &c);
    lowercase_vowel = (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u');
    uppercase_vowel = (c == 'A' || c == 'E' || c == 'I' || c == 'O' || c == 'U');
    if (lowercase_vowel || uppercase_vowel)
        printf("%c is a vowel.", c);
    else
        printf("%c is a consonant.", c);
    return 0;
}

To Find the Largest Number Among Three Numbers

CODE

    int n1, n2, n3;
    printf("Enter three different numbers: ");
    scanf("%d %d %d", &n1, &n2, &n3);
    if (n1 >= n2 && n1 >= n3)
        printf("%d is the largest number.\n", n1);
    else if (n2 >= n1 && n2 >= n3)
        printf("%d is the largest number.\n", n2);
    else
        printf("%d is the largest number.\n", n3);
    return 0;
}

Second program:

strlwr() - Convert to lowercase

CODE

#include <stdio.h>
#include <string.h>
int main(){
    char a[100], b[100];
    printf("Enter the first string: ");
    scanf("%99s", a);
    printf("Enter the second string: ");
    scanf("%99s", b);
    for (int i = 0; i < strlen(a); i++) {
        if (a[i] >= 'A' && a[i] <= 'Z') {
            a[i] = a[i] + ('a' - 'A');
        }
    }
    for (int i = 0; i < strlen(b); i++) {
        if (b[i] >= 'A' && b[i] <= 'Z') {
            b[i] = b[i] + ('a' - 'A');
        }
    }
    printf("\nString 1 in lowercase: %s\n", a);
    printf("String 2 in lowercase: %s\n", b);

    return 0;
}

strupr() - Convert to uppercase

CODE

#include<stdio.h>
#include <string.h>
int main(){
    char a[100], b[100];
    printf("Enter the first string: ");
    scanf("%99s", a);  // Read input safely for a
    printf("Enter the second string: ");
    scanf("%99s", b);  // Read input safely for b
    for (int i = 0; i < strlen(a); i++) {
        a[i] = toupper(a[i]);
    }
    for (int i = 0; i < strlen(b); i++) {
        b[i] = toupper(b[i]);
    }
    printf("\nString 1 in uppercase: %s\n", a);
    printf("String 2 in uppercase: %s\n", b);
    return 0;
}

CODE

    printf("Enter first string: ");
